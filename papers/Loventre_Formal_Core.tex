\section{Efficiency Axiom}

\textbf{Axiom EfficientBound}: \\
\textit{forall (traj : list C), efficient traj} \\
\textit{-> exists B : Val,} \\
\textit{    forall f, (f = kappa $\vee$ f = entropy $\vee$ f = compactness)} \\
\textit{    -> forall c1 c2, In c1 traj $\to$ In c2 traj $\to$ leVal (delta f c1 c2) B.}

\section{Structural Horizon Definition}

\textbf{Definition Horizon}: \\
\textit{Horizon c : Prop} \\
\textit{:= forall traj : list C,} \\
\textit{    In c traj -> (exists cT, In cT traj /\ Target cT) -> ~ efficient traj.}

\section{Lemma L3 — Efficiency Obstruction}

\textbf{Theorem NoEfficientCrossing}: \\
\textit{forall c : C, Horizon c ->} \\
\textit{    forall traj : list C, In c traj -> (exists cT, In cT traj /\ Target cT) -> ~ efficient traj.} \\
\textbf{Proof}: \\
\textit{intros c Hhor traj Hc HT Heff.} \\
\textit{unfold Horizon in Hhor.} \\
\textit{apply (Hhor traj); assumption.} \\
\textit{Qed.}
